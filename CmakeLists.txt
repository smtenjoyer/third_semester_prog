cmake_minimum_required(VERSION 3.5.0)
project(ip_check VERSION 0.1.0 LANGUAGES C CXX)

# Указываем стандарт C++
set(CMAKE_CXX_STANDARD 11)
set(CMAKE_CXX_STANDARD_REQUIRED True)

# Добавляем исполняемый файл для основного приложения
add_executable(ip_check ipfilter.cpp main.cpp)

# Добавляем поддиректорию для тестов
add_subdirectory(gtests)

# Указываем настройки для сборки пакетов
set(CPACK_PROJECT_NAME ${PROJECT_NAME})
set(CPACK_PROJECT_VERSION ${PROJECT_VERSION})
include(CPack)
